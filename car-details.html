document.addEventListener('DOMContentLoaded', function () { // Utility Function to Format Price const formatPrice = (price) => { // Remove any existing currency symbols and non-numeric characters const numericPrice = Number(price.toString().replace(/[^0-9.]/g, '')); return `${numericPrice.toLocaleString()} DHS`; }; // Get car ID from URL const urlParams = new URLSearchParams(window.location.search); const carId = urlParams.get('id'); if (!carId) { document.body.innerHTML = `

Invalid Request

No car ID was provided in the URL.

Return to Inventory
`; return; } // Load car data fetch('cars.json') .then(response => { if (!response.ok) throw new Error('Failed to load cars.'); return response.json(); }) .then(data => { const car = data.cars.find(item => item.id == carId); if (!car) { document.body.innerHTML = `

Car Not Found

The car you're looking for does not exist.

Return to Inventory
`; return; } // Populate page document.getElementById('car-title').textContent = `${car.make} ${car.model} (${car.year})`; // Format and display the price with "DHS" document.getElementById('car-price').textContent = formatPrice(car.price); const badge = document.getElementById('availability-badge'); badge.textContent = car.available ? 'Available' : 'Sold'; badge.classList.add(car.available ? 'available' : 'sold'); document.getElementById('car-description').textContent = car.description || 'No additional description available.'; // Main image const mainImg = document.getElementById('main-image'); mainImg.src = car.image; mainImg.alt = `${car.make} ${car.model}`; // Thumbnails const thumbContainer = document.getElementById('thumbnails'); if (car.images && car.images.length > 0) { car.images.forEach((img, index) => { const thumb = document.createElement('img'); thumb.src = img; thumb.className = 'thumbnail'; thumb.onclick = () => mainImg.src = img; thumbContainer.appendChild(thumb); }); } else { thumbContainer.innerHTML = '

No additional images available.

'; } // Specs const specs = [ { label: 'Make', value: car.make }, { label: 'Model', value: car.model }, { label: 'Year', value: car.year }, { label: 'Mileage', value: car.mileage || 'N/A' }, { label: 'Transmission', value: car.transmission || 'N/A' }, { label: 'Fuel Type', value: car.fuelType || 'N/A' } ]; const specsGrid = document.getElementById('specs-grid'); specs.forEach(spec => { const item = document.createElement('div'); item.className = 'spec-item'; item.innerHTML = `${spec.label}: ${spec.value}`; specsGrid.appendChild(item); }); // Contact buttons document.getElementById('whatsapp-btn').onclick = () => { window.open(`https://wa.me/${car.contact.whatsapp}?text=Interested in ${car.make} ${car.model} (ID: ${car.id})`); }; document.getElementById('call-btn').onclick = () => { window.location.href = `tel:${car.contact.phone}`; }; }) .catch(error => { console.error('Error loading cars:', error); document.body.innerHTML = `

Error Loading Data

An unexpected error occurred. Please try again later.

Return to Inventory
`; }); });
